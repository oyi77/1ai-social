# Remotion 3D with Three.js Rule

## Setup

```bash
npm install @remotion/three three @types/three
```

## Basic ThreeCanvas

```tsx
import { ThreeCanvas } from '@remotion/three';
import { AbsoluteFill, useCurrentFrame, useVideoConfig } from 'remotion';
import { useRef } from 'react';
import * as THREE from 'three';

export const ThreeDScene = () => {
  const frame = useCurrentFrame();
  const { width, height } = useVideoConfig();

  return (
    <AbsoluteFill>
      <ThreeCanvas width={width} height={height}>
        <ambientLight intensity={0.5} />
        <pointLight position={[10, 10, 10]} />
        <RotatingBox frame={frame} />
      </ThreeCanvas>
    </AbsoluteFill>
  );
};

const RotatingBox: React.FC<{ frame: number }> = ({ frame }) => {
  const rotation = (frame / 30) * Math.PI; // one rotation per second

  return (
    <mesh rotation={[rotation, rotation * 0.7, 0]}>
      <boxGeometry args={[2, 2, 2]} />
      <meshStandardMaterial color="#6366f1" />
    </mesh>
  );
};
```

## Using spring() with Three.js

```tsx
import { spring, useCurrentFrame, useVideoConfig } from 'remotion';

const AnimatedSphere: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps } = useVideoConfig();

  const scale = spring({ frame, fps, config: { damping: 10 } });
  const rotY = (frame / fps) * Math.PI * 2 * 0.5; // 0.5 rotations/sec

  return (
    <mesh scale={[scale, scale, scale]} rotation={[0, rotY, 0]}>
      <sphereGeometry args={[1.5, 32, 32]} />
      <meshStandardMaterial color="#22c55e" wireframe />
    </mesh>
  );
};
```

## React Three Fiber (R3F) — More Expressive

```bash
npm install @react-three/fiber @react-three/drei
```

```tsx
import { ThreeCanvas } from '@remotion/three';
import { OrbitControls, Text3D, Environment } from '@react-three/drei';

export const R3FScene = () => {
  const frame = useCurrentFrame();
  const { width, height, fps } = useVideoConfig();

  return (
    <AbsoluteFill>
      <ThreeCanvas width={width} height={height}>
        <Environment preset="city" />
        <ambientLight intensity={0.3} />

        <Text3D
          font="/fonts/helvetiker_regular.typeface.json"
          size={0.8}
          height={0.2}
          rotation={[0, (frame / fps) * 0.5, 0]}
        >
          Remotion 3D
          <meshStandardMaterial color="#6366f1" metalness={0.8} roughness={0.2} />
        </Text3D>
      </ThreeCanvas>
    </AbsoluteFill>
  );
};
```

## Particle System

```tsx
import { useMemo } from 'react';
import { interpolate, useCurrentFrame } from 'remotion';

const Particles: React.FC<{ count?: number }> = ({ count = 200 }) => {
  const frame = useCurrentFrame();

  const particles = useMemo(() => {
    return Array.from({ length: count }, (_, i) => ({
      x: (Math.random() - 0.5) * 10,
      y: (Math.random() - 0.5) * 10,
      z: (Math.random() - 0.5) * 10,
      speed: 0.01 + Math.random() * 0.02,
      phase: Math.random() * Math.PI * 2,
    }));
  }, [count]);

  return (
    <>
      {particles.map((p, i) => (
        <mesh
          key={i}
          position={[
            p.x,
            p.y + Math.sin(frame * p.speed + p.phase) * 0.5,
            p.z,
          ]}
        >
          <sphereGeometry args={[0.03, 8, 8]} />
          <meshBasicMaterial color="#4f46e5" />
        </mesh>
      ))}
    </>
  );
};
```

## Camera Animation

```tsx
import { PerspectiveCamera } from '@react-three/drei';
import { spring, interpolate } from 'remotion';

const AnimatedCamera: React.FC<{ frame: number; fps: number }> = ({ frame, fps }) => {
  const progress = spring({ frame, fps, config: { damping: 80 } });
  const z = interpolate(progress, [0, 1], [8, 4]); // zoom in

  return <PerspectiveCamera makeDefault position={[0, 0, z]} fov={50} />;
};
```

## Important Notes for 3D in Remotion

1. **Never use `useFrame` from R3F** — it drives animations via requestAnimationFrame, not Remotion's frame system. Use `useCurrentFrame()` instead.
2. **All animations must be driven by `frame`** — pass frame as a prop to 3D components.
3. **ThreeCanvas requires `width` and `height`** from `useVideoConfig()`.
4. **Avoid heavy meshes** — each frame is rendered as a screenshot; complex geometries slow rendering.
5. **Use `useMemo` for geometry and materials** — prevents recreation on every frame.
